285// Prevent 64-bit integer overflow when computing a timestamp by using a trick
286// from Sokol:
287// https://github.com/floooh/sokol/blob/189843bf4f86969ca4cc4b6d94e793a37c5128a7/sokol_time.h#L204
288static UBENCH_INLINE ubench_int64_t ubench_mul_div(const ubench_int64_t value,
289 const ubench_int64_t numer,
290 const ubench_int64_t denom) {
291 const ubench_int64_t q = value / denom;
292 const ubench_int64_t r = value % denom;
293 return q * numer + r * numer / denom;
294}

296static UBENCH_INLINE ubench_int64_t ubench_ns(void) {
297#if defined(_MSC_VER) || defined(__MINGW64__) || defined(__MINGW32__)
298 ubench_large_integer counter;
299 ubench_large_integer frequency;
300 QueryPerformanceCounter(&counter);
301 QueryPerformanceFrequency(&frequency);
302 return ubench_mul_div(counter.QuadPart, 1000000000, frequency.QuadPart);
303#elif defined(__linux__) && defined(__STRICT_ANSI__)
304 return ubench_mul_div(clock(), 1000000000, CLOCKS_PER_SEC);
305#elif defined(__linux__) || defined(__FreeBSD__) || defined(__OpenBSD__) || \
306 defined(__NetBSD__) || defined(__DragonFly__) || defined(__sun__) || \
307 defined(__HAIKU__)
308 struct timespec ts;
309#if defined(__STDC_VERSION__) && (__STDC_VERSION__ >= 201112L) && \
310 !defined(__HAIKU__)
311 timespec_get(&ts, TIME_UTC);
312#else
313 const clockid_t cid = CLOCK_REALTIME;
314#if defined(UBENCH_USE_CLOCKGETTIME)
315 clock_gettime(cid, &ts);
316#else
317 syscall(SYS_clock_gettime, cid, &ts);
318#endif
319#endif
320 return UBENCH_CAST(ubench_int64_t, ts.tv_sec) * 1000 * 1000 * 1000 +
321 ts.tv_nsec;
322#elif __APPLE__
323 return UBENCH_CAST(ubench_int64_t, clock_gettime_nsec_np(CLOCK_UPTIME_RAW));
324#elif __EMSCRIPTEN__
325 return emscripten_performance_now() * 1000000.0;
326#else
327#error Unsupported platform!
328#endif
329}

502static UBENCH_INLINE int ubench_should_filter(const char *filter,
503 const char *benchmark);
504int ubench_should_filter(const char *filter, const char *benchmark) {
505 if (filter) {
506 const char *filter_cur = filter;
507 const char *benchmark_cur = benchmark;
508 const char *filter_wildcard = UBENCH_NULL;
509
510 while (('\0' != *filter_cur) && ('\0' != *benchmark_cur)) {
511 if ('*' == *filter_cur) {
512 /* store the position of the wildcard */
513 filter_wildcard = filter_cur;
514
515 /* skip the wildcard character */
516 filter_cur++;
517
518 while (('\0' != *filter_cur) && ('\0' != *benchmark_cur)) {
519 if ('*' == *filter_cur) {
520 /*
521 we found another wildcard (filter is something like *foo*) so we
522 exit the current loop, and return to the parent loop to handle
523 the wildcard case
524 */
525 break;
526 } else if (*filter_cur != *benchmark_cur) {
527 /* otherwise our filter didn't match, so reset it */
528 filter_cur = filter_wildcard;
529 }
530
531 /* move benchmark along */
532 benchmark_cur++;
533
534 /* move filter along */
535 filter_cur++;
536 }
537
538 if (('\0' == *filter_cur) && ('\0' == *benchmark_cur)) {
539 return 0;
540 }
541
542 /* if the benchmarks have been exhausted, we don't have a match! */
543 if ('\0' == *benchmark_cur) {
544 return 1;
545 }
546 } else {
547 if (*benchmark_cur != *filter_cur) {
548 /* benchmark doesn't match filter */
549 return 1;
550 } else {
551 /* move our filter and benchmark forward */
552 benchmark_cur++;
553 filter_cur++;
554 }
555 }
556 }
557
558 if (('\0' != *filter_cur) ||
559 (('\0' != *benchmark_cur) &&
560 ((filter == filter_cur) || ('*' != filter_cur[-1])))) {
561 /* we have a mismatch! */
562 return 1;
563 }
564 }
565
566 return 0;
567}

569static UBENCH_INLINE int ubench_strncmp(const char *a, const char *b,
570 size_t n) {
571 /* strncmp breaks on Wall / Werror on gcc/clang, so we avoid using it */
572 unsigned i;
573
574 for (i = 0; i < n; i++) {
575 if (a[i] < b[i]) {
576 return -1;
577 } else if (a[i] > b[i]) {
578 return 1;
579 }
580 }
581
582 return 0;
583}

599static UBENCH_INLINE int ubench_main(int argc, const char *const argv[]);
600int ubench_main(int argc, const char *const argv[]) {
601 ubench_uint64_t failed = 0;
602 size_t index = 0;
603 size_t *failed_benchmarks = UBENCH_NULL;
604 size_t failed_benchmarks_length = 0;
605 const char *filter = UBENCH_NULL;
606 ubench_uint64_t ran_benchmarks = 0;
607
608 enum colours { RESET, GREEN, RED };
609
610 const int use_colours = UBENCH_COLOUR_OUTPUT();
611 const char *colours[] = {"\033[0m", "\033[32m", "\033[31m"};
612 if (!use_colours) {
613 for (index = 0; index < sizeof colours / sizeof colours[0]; index++) {
614 colours[index] = "";
615 }
616 }
617
618 /* loop through all arguments looking for our options */
619 for (index = 1; index < UBENCH_CAST(size_t, argc); index++) {
620 /* Informational switches */
621 const char help_str[] = "--help";
622 const char list_str[] = "--list-benchmarks";
623 /* Benchmark config switches */
624 const char filter_str[] = "--filter=";
625 const char output_str[] = "--output=";
626 const char confidence_str[] = "--confidence=";
627
628 if (0 == ubench_strncmp(argv[index], help_str, strlen(help_str))) {
629 printf("ubench.h - the single file benchmarking solution for C/C++!\n"
630 "Command line Options:\n");
631 printf(" --help Show this message and exit.\n"
632 " --filter=<filter> Filter the benchmarks to run (EG. "
633 "MyBench*.a would run MyBenchmark.a but not MyBenchmark.b).\n"
634 " --list-benchmarks List benchmarks, one per line. "
635 "Output names can be passed to --filter.\n"
636 " --output=<output> Output a CSV file of the results.\n"
637 " --confidence=<confidence> Change the confidence cut-off for a "
638 "failed test. Defaults to 2.5%%\n");
639 goto cleanup;
640 } else if (0 ==
641 ubench_strncmp(argv[index], filter_str, strlen(filter_str))) {
642 /* user wants to filter what benchmarks run! */
643 filter = argv[index] + strlen(filter_str);
644 } else if (0 ==
645 ubench_strncmp(argv[index], output_str, strlen(output_str))) {
646 ubench_state.output =
647 ubench_fopen(argv[index] + strlen(output_str), "w+");
648 } else if (0 == ubench_strncmp(argv[index], list_str, strlen(list_str))) {
649 for (index = 0; index < ubench_state.benchmarks_length; index++) {
650 UBENCH_PRINTF("%s\n", ubench_state.benchmarks[index].name);
651 }
652
653 /* when printing the benchmark list, don't actually run the benchmarks */
654 goto cleanup;
655 } else if (0 == ubench_strncmp(argv[index], confidence_str,
656 strlen(confidence_str))) {
657 /* user wants to specify a different confidence */
658 ubench_state.confidence = atof(argv[index] + strlen(confidence_str));
659
660 /* must be between 0 and 100 */
661 if ((ubench_state.confidence < 0) || (ubench_state.confidence > 100)) {
662 fprintf(stderr,
663 "Confidence must be in the range [0..100] (you specified %f)\n",
664 ubench_state.confidence);
665 goto cleanup;
666 }
667 }
668 }
669
670 for (index = 0; index < ubench_state.benchmarks_length; index++) {
671 if (ubench_should_filter(filter, ubench_state.benchmarks[index].name)) {
672 continue;
673 }
674
675 ran_benchmarks++;
676 }
677
678 printf("%s[==========]%s Running %" UBENCH_PRIu64 " benchmarks.\n",
679 colours[GREEN], colours[RESET],
680 UBENCH_CAST(ubench_uint64_t, ran_benchmarks));
681
682 if (ubench_state.output) {
683 fprintf(ubench_state.output,
684 "name, mean (ns), stddev (%%), confidence (%%)\n");
685 }
686
687 for (index = 0; index < ubench_state.benchmarks_length; index++) {
688 int result = 1;
689 size_t mndex = 0;
690 ubench_int64_t best_avg_ns = 0;
691 double best_deviation = 0;
692 double best_confidence = 101.0;
693 struct ubench_run_state_s ubs;
694
695#define UBENCH_MIN_ITERATIONS 10
696#define UBENCH_MAX_ITERATIONS 500
697 ubench_int64_t iterations = 10;
698 const ubench_int64_t max_iterations = UBENCH_MAX_ITERATIONS;
699 const ubench_int64_t min_iterations = UBENCH_MIN_ITERATIONS;
700 /* Add one extra timestamp slot, as we save times between runs and time
701 * after exiting the last one */
702 ubench_int64_t ns[UBENCH_MAX_ITERATIONS + 1];
703#undef UBENCH_MAX_ITERATIONS
704#undef UBENCH_MIN_ITERATIONS
705
706 if (ubench_should_filter(filter, ubench_state.benchmarks[index].name)) {
707 continue;
708 }
709
710 printf("%s[ RUN ]%s %s\n", colours[GREEN], colours[RESET],
711 ubench_state.benchmarks[index].name);
712
713 ubs.ns = ns;
714 ubs.size = 1;
715 ubs.sample = 0;
716
717 /* Time once to work out the base number of iterations to use. */
718 ubench_state.benchmarks[index].func(&ubs);
719
720 iterations = (100 * 1000 * 1000) / ((ns[1] <= ns[0]) ? 1 : ns[1] - ns[0]);
721 iterations = iterations < min_iterations ? min_iterations : iterations;
722 iterations = iterations > max_iterations ? max_iterations : iterations;
723
724 for (mndex = 0; (mndex < 100) && (result != 0); mndex++) {
725 ubench_int64_t kndex = 0;
726 ubench_int64_t avg_ns = 0;
727 double deviation = 0;
728 double confidence = 0;
729
730 iterations = iterations * (UBENCH_CAST(ubench_int64_t, mndex) + 1);
731 iterations = iterations > max_iterations ? max_iterations : iterations;
732
733 ubs.sample = 0;
734 ubs.size = iterations;
735 ubench_state.benchmarks[index].func(&ubs);
736
737 /* Calculate benchmark run-times */
738 for (kndex = 0; kndex < iterations; kndex++) {
739 ns[kndex] = ns[kndex + 1] - ns[kndex];
740 }
741
742 for (kndex = 0; kndex < iterations; kndex++) {
743 avg_ns += ns[kndex];
744 }
745
746 avg_ns /= iterations;
747
748 for (kndex = 0; kndex < iterations; kndex++) {
749 const double v = UBENCH_CAST(double, ns[kndex] - avg_ns);
750 deviation += v * v;
751 }
752
753 deviation = sqrt(deviation / UBENCH_CAST(double, iterations));
754
755 /* Confidence is the 99% confidence index - whose magic value is 2.576. */
756 confidence = 2.576 * deviation / sqrt(UBENCH_CAST(double, iterations));
757 confidence = (confidence / UBENCH_CAST(double, avg_ns)) * 100.0;
758
759 deviation = (deviation / UBENCH_CAST(double, avg_ns)) * 100.0;
760
761 /* If we've found a more confident solution, use that. */
762 result = confidence > ubench_state.confidence;
763
764 /* If the deviation beats our previous best, record it. */
765 if (confidence < best_confidence) {
766 best_avg_ns = avg_ns;
767 best_deviation = deviation;
768 best_confidence = confidence;
769 }
770 }
771
772 if (result) {
773 printf("confidence interval %f%% exceeds maximum permitted %f%%\n",
774 best_confidence, ubench_state.confidence);
775 }
776
777 if (ubench_state.output) {
778 fprintf(ubench_state.output, "%s, %" UBENCH_PRId64 ", %f, %f,\n",
779 ubench_state.benchmarks[index].name, best_avg_ns, best_deviation,
780 best_confidence);
781 }
782
783 {
784 const char *const colour = (0 != result) ? colours[RED] : colours[GREEN];
785 const char *const status =
786 (0 != result) ? "[ FAILED ]" : "[ OK ]";
787 const char *unit = "us";
788
789 if (0 != result) {
790 const size_t failed_benchmark_index = failed_benchmarks_length++;
791 failed_benchmarks = UBENCH_PTR_CAST(
792 size_t *, realloc(UBENCH_PTR_CAST(void *, failed_benchmarks),
793 sizeof(size_t) * failed_benchmarks_length));
794 failed_benchmarks[failed_benchmark_index] = index;
795 failed++;
796 }
797
798 printf("%s%s%s %s (mean ", colour, status, colours[RESET],
799 ubench_state.benchmarks[index].name);
800
801 for (mndex = 0; mndex < 2; mndex++) {
802 if (best_avg_ns <= 1000000) {
803 break;
804 }
805
806 /* If the average is greater than a million, we reduce it and change the
807 unit we report. */
808 best_avg_ns /= 1000;
809
810 switch (mndex) {
811 case 0:
812 unit = "ms";
813 break;
814 case 1:
815 unit = "s";
816 break;
817 }
818 }
819
820 printf("%" UBENCH_PRId64 ".%03" UBENCH_PRId64
821 "%s, confidence interval +- %f%%)\n",
822 best_avg_ns / 1000, best_avg_ns % 1000, unit, best_confidence);
823 }
824 }
825
826 printf("%s[==========]%s %" UBENCH_PRIu64 " benchmarks ran.\n",
827 colours[GREEN], colours[RESET], ran_benchmarks);
828 printf("%s[ PASSED ]%s %" UBENCH_PRIu64 " benchmarks.\n", colours[GREEN],
829 colours[RESET], ran_benchmarks - failed);
830
831 if (0 != failed) {
832 printf("%s[ FAILED ]%s %" UBENCH_PRIu64 " benchmarks, listed below:\n",
833 colours[RED], colours[RESET], failed);
834 for (index = 0; index < failed_benchmarks_length; index++) {
835 printf("%s[ FAILED ]%s %s\n", colours[RED], colours[RESET],
836 ubench_state.benchmarks[failed_benchmarks[index]].name);
837 }
838 }
839
840cleanup:
841 for (index = 0; index < ubench_state.benchmarks_length; index++) {
842 free(UBENCH_PTR_CAST(void *, ubench_state.benchmarks[index].name));
843 }
844
845 free(UBENCH_PTR_CAST(void *, failed_benchmarks));
846 free(UBENCH_PTR_CAST(void *, ubench_state.benchmarks));
847
848 if (ubench_state.output) {
849 fclose(ubench_state.output);
850 }
851
852 return UBENCH_CAST(int, failed);
853}
